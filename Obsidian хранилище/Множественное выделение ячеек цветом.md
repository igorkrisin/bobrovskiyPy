Для поддержки множественного выделения ячеек в таблице вы можете использовать свойство allowsMultipleSelection вашего UITableView и методы делегата tableView(_:didSelectRowAt:) и tableView(_:didDeselectRowAt:). Вот пример:

```swift
import UIKit

// Создаем подкласс UITableViewCell
class MyTableViewCell: UITableViewCell {
    // Добавляем необходимые элементы интерфейса
    let titleLabel: UILabel = {
        let label = UILabel()
        label.translatesAutoresizingMaskIntoConstraints = false
        return label
    }()
    
    override init(style: UITableViewCell.CellStyle, reuseIdentifier: String?) {
        super.init(style: style, reuseIdentifier: reuseIdentifier)
        
        // Добавляем titleLabel к ячейке
        contentView.addSubview(titleLabel)
        
        // Настраиваем ограничения для titleLabel
        NSLayoutConstraint.activate([
            titleLabel.leadingAnchor.constraint(equalTo: contentView.leadingAnchor, constant: 16),
            titleLabel.trailingAnchor.constraint(equalTo: contentView.trailingAnchor, constant: -16),
            titleLabel.topAnchor.constraint(equalTo: contentView.topAnchor, constant: 8),
            titleLabel.bottomAnchor.constraint(equalTo: contentView.bottomAnchor, constant: -8)
        ])
    }
    
    required init?(coder aDecoder: NSCoder) {
        fatalError("init(coder:) has not been implemented")
    }
}

// Ваш контроллер, содержащий UITableView
class MyTableViewController: UITableViewController {
    override func viewDidLoad() {
        super.viewDidLoad()
        
        // Регистрируем класс MyTableViewCell для использования в таблице
        tableView.register(MyTableViewCell.self, forCellReuseIdentifier: "MyCell")
        
        // Разрешаем множественное выделение ячеек
        tableView.allowsMultipleSelection = true
    }
    
    // Реализуем методы делегата и источника данных для таблицы
    override func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        // Возвращаем количество строк в секции
        return 10 // Например, 10 строк
    }
    
    override func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        // Создаем ячейку и настраиваем ее содержимое
        let cell = tableView.dequeueReusableCell(withIdentifier: "MyCell", for: indexPath) as! MyTableViewCell
        
        // Например, устанавливаем текст для titleLabel
        cell.titleLabel.text = "Строка \(indexPath.row + 1)"
        
        return cell
    }
    
    override func tableView(_ tableView: UITableView, didSelectRowAt indexPath: IndexPath) {
        // Получаем выбранную ячейку
        guard let cell = tableView.cellForRow(at: indexPath) else { return }
        
        // Устанавливаем цвет фона для выбранной ячейки
        cell.contentView.backgroundColor = .yellow
    }
    
    override func tableView(_ tableView: UITableView, didDeselectRowAt indexPath: IndexPath) {
        // Получаем выбранную ячейку
        guard let cell = tableView.cellForRow(at: indexPath) else { return }
        
        // Возвращаем прозрачный цвет фона для снятой выделенной ячейки
        cell.contentView.backgroundColor = .clear
    }
}
```

Этот код позволит пользователю выделять несколько ячеек одновременно. Когда ячейка выбрана, ее фон становится желтым, а когда она снимается с выбора, фон возвращается к прозрачному цвету.
